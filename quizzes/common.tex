\def \makeDashedLine{
  \begin{tikzpicture}[remember picture, overlay]
    \draw[shift={(current page.west |- 0,0)}, lightgray, very thin, dashed]
    (1in,-0.5\baselineskip) -- (7.5in,-0.5\baselineskip);
  \end{tikzpicture}
}

% <https://en.wikipedia.org/wiki/Ruled_paper>
% - college ruled paper has a line spacing of 9/32in (7.1mm); this gives the
%   students 32 rows and 30 columns
\def \makePageQuadrilleRuled{
  \begin{tikzpicture}[remember picture, overlay]
    \draw[shift=(current page.center), step=7.1mm, lightgray, very thin, dashed]
    (current page.north west) grid (current page.south east);
  \end{tikzpicture}
}

\def \showCodeTop #1#2{
  \def\codeFile{\docCodeDir/.gen.#1.cpp.section.#2}
  \IfFileExists{\docCodeDir/.gen.#1.output.section.#2}
               {\def\outputFile{\docCodeDir/.gen.#1.output.section.#2}}
               {\def\outputFile{\docCodeDir/.gen.#1.output.section.all}}
  \par
  \inputminted{cpp}{\codeFile}
  \vspace{0.5\parskip}
  \inputminted[label=Output]{text}{\outputFile}
  \par
}

  \begin{minipage}[t]{0.5\linewidth} \vspace{0ex}
    \inputminted[label=Output]{text}{\outputFile}
  \end{minipage}
  \par
}

\def \evaluateCodeTop #1#2{
  \def\codeFile{\docCodeDir/.gen.#1.cpp.section.#2}
  \IfFileExists{\docCodeDir/.gen.#1.output.section.#2}
               {\def\outputFile{\docCodeDir/.gen.#1.output.section.#2}}
               {\def\outputFile{\docCodeDir/.gen.#1.output.section.all}}
  \par
  \inputminted{cpp}{\codeFile}
  \textQuestion{
    \inputminted[label={\normalsize{}Output},fontsize=\Large]{text}
      {\outputFile.blank}
  }
  \textAnswer{
    \inputminted[label={\normalsize{}Output},fontsize=\Large]{text}
      {\outputFile}
  }
  \par
}

\def \correctCodeTop #1#2{
  \def\codeFile{\docCodeDir/.gen.#1.cpp.section.#2}
  \def\outputFile{\docCodeDir/.gen.#1.output.section.#2}
  \par
  \inputminted[label={\normalsize{}Incorrect Code},fontsize=\Large]{cpp}{\codeFile--bad}
  \par
  \textQuestion{
    \IfFileExists{\outputFile--good}
      {\inputminted[label={Intended Output}]{text}{\outputFile--good}}
      {}
  }
  \textQuestion{
    \IfFileExists{\outputFile--bad}
      {\inputminted[label={Actual Output}]{text}{\outputFile--bad}}
      {}
  }
  \textAnswer{\inputminted[label={Corrections}]{diff}{\codeFile--diff}}
  \par
}
